create table account_master(account_number VARCHAR(6),customer_number VARCHAR(6),branch_id VARCHAR(6),opening_balance INT(7),account_opening_date DATE,account_type VARCHAR(10),account_status VARCHAR(10),PRIMARY KEY(account_number),FOREIGN KEY(customer_number,branch_id) REFERENCES loan_details(customer_number,branch_id));
